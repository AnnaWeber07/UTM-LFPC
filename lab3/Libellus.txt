PARTES WINE (VARIABILIS VINO)
{
VINO = ROSA;
SI (VINO == ALBUM)
{
REDITIO VERUM;
}
ALITER
{
REDITIO FALSUS;
}
}

PARTES COMPUTARE(NUMERUS PRIMIS, NUMERUS SECUNDO)
{
NUMERUS EVENTUM = PRIMIS - SECUNDO;
SI (EVENTUM > 0)
{
REDITIO "POSITIVUM";
}
ALITER
{
REDITIO "NEGATIVUS";
}
}

PARTES PRIMOR()
{
VARIABILIS VINO;
NUMERUS PRIMIS = 70;
NUMERUS SECUNDO = 40;

NOTA(COMPUTARE(PRIMIS, SECUNDO));
NOTA("HOC EST LATIMUN LEXER");
}
